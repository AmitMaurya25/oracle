For testing purpose we will create once user and table and do insert operation:

CREATE USER TEST1 IDENTIFIED BY TEST DEFAULT TABLESPACE "USERS" TEMPORARY TABLESPACE "TEMP";

ALTER USER TEST1 QUOTA UNLIMITED ON USERs;

GRANT CONNECT TO TEST1;

GRANT RESOURCE TO TEST1;

create table TEST1.DEMO ("id" number , "text" varchar2(15), "number" number) ;

declare
 type people_array is varray(12) of varchar(15);
 people people_array := people_array( 'Marc', 'Bill', 'George', 'Eliot', 'Matt', 'Trey', 'Tracy','Greg', 'Steve', 'Kristina', 'Katie', 'Jeff' );
 people_count number :=people.COUNT;
 n number;
begin
 for i in 0..1e6 loop
  n:=trunc(dbms_random.value(0,100000));
  insert into TEST1.DEMO values( i , people( dbms_random.value(1,people_count) ) , n );
 end loop;
 commit;
end;
/

select count(0) from TEST1.DEMO;
